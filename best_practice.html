CODE

#define F_CPU 8000000UL
#include <avr/io.h>
#include <avr/interrupt.h>
#include <util/delay.h>

char dig[] = {0b11000000, 0b11111001, 0b10100100, 0b10110000, 0b10011001, 0b10010010, 0b10000010, 0b11111000, 0b10000000, 0b10010000, 0b01111111};
char razr[] = {0b11111110, 0b11111101, 0b11111011, 0b11110111};
unsigned char display[2];
unsigned int a=0, i;
unsigned char dig1, dig2, dig3, dig4;


void write_display(unsigned char *data)
{
unsigned char mask,i;

for(i = 0; i < 2; i++)
{
 mask = 0x80;
 
 for(char k = 0; k < 8; k++)
 {
  // сравниваем каждый бит с единицей
  if(data[i] & mask)
  {
   PORTD &=~(1 << PD5); // DATA 1
   PORTD |= (1 << PD6); // CLK 1
   PORTD &= ~(1 << PD6); // CLK 0
  }
  else
  {
   PORTD |=(1 << PD5); // DATA 0
   PORTD |= (1 << PD6); // CLK 1
   PORTD &= ~(1 << PD6); // CLK 0
  }
  mask = mask >> 1; // сдвигаем биты
 }
}
// защелкиваем регистр
PORTD |= (1 << PD4);
PORTD &= ~(1 << PD4);
}


ISR (TIMER1_COMPA_vect)
{
 dig1=dig[a % 10000 / 1000];

 display[0] =  razr[3];
 display[1] =  dig4;

 display[1] =  dig2;

 write_display(display);  // и посылается на 74hc595
}

При входе в прерывание сначала тушим всё общие выводы сегментов:

Вот так например:
CODE

PORTD &= ~((1<<4)|(1<<5)|(1<<6)|(1<<7)); // Не знаю, правильно ли, но вот так &=~0b11110000 будет работать точно.


Потом передаём данные на индикатор:
CODE

SPDR = dig[i]; // Привязываем к счётчику i, смысла использовать IF для этого нет!
while( !( SPSR & ( 1 << SPIF ) ) );
SPI_PORT &= ~(1 << SPI_SS ); // Дергаем строб 74hc595
SPI_PORT |= ( 1 << SPI_SS );


Данные переданы, теперь можно зажечь соответствующий сегмент:
Вот так например:
CODE

PORTD |= (1<<i+4); //Привязываем к счётчику i, незачем ему простаивать, +4 это прибавка, чтобы значении 0 счётчика получить число 4, то есть это равносильно (1<<4), при значении счётчика 1 равносильно (1<<5) и т.д.


Икрементируем счётчик:
CODE

i++;


Затем проверим счётчик на пределы, если что, сбросим его:
CODE

if(i > 3) i = 0;


Всё!

P.S. Незабываем, что dig[] это массив, поэтому надо его объявить unsigned char dig[4];
В основной программе туда данные можно теперь складывать при помощи цикла, а не по одному, как было раньше с отдельными переменными.

Динамическая индикация:
CODE

#include <avr/io.h>
#include <avr/interrupt.h>
#include <avr/pgmspace.h>
#include <util/delay.h>

unsigned int in = 0;

unsigned char i;
unsigned char DisplayBuffer[4];

unsigned int div_table[] PROGMEM =
{
1000,
100,
10,
1
};

unsigned char dec_table[] PROGMEM =
{
0b00111111, //'0'  #0
0b00000110, //'1'  #1
0b01011011, //'2'  #2
0b01001111, //'3'  #3
0b01100110, //'4'  #4
0b01101101, //'5'  #5
0b01111101, //'6'  #6
0b00000111, //'7'  #7
0b01111111, //'8'  #8
0b01101111, //'9'  #9
};

/*
// Общий анод
ISR(TIMER0_OVF_vect)
{
PORTB &= ~((1<<PB0)|(1<<PB1)|(1<<PB2)|(1<<PB3));
SPDR = ~DisplayBuffer[i];
while(!(SPSR&(1<<SPIF)));
PORTB |=  (1<<PB4);
PORTB &= ~(1<<PB4);
PORTB |=  (1<<i);
i++;
if(i>3) i=0;
}
*/

// Общий катод
ISR(TIMER0_OVF_vect)
{
PORTB |=  (1<<PB0)|(1<<PB1)|(1<<PB2)|(1<<PB3);
SPDR = DisplayBuffer[i];
while(!(SPSR&(1<<SPIF)));
PORTB |=  (1<<PB4);
PORTB &= ~(1<<PB4);
PORTB &= ~(1<<i);
i++;
if(i>3) i=0;
}

void update(void)
{
unsigned char counter;
unsigned char offset = 0;

/* Надо подумать как можно запихнуть это в цикл */
if(in >= 1000) // default state
{
goto L0; // Да и метки тут нежелательны
}

if(in >= 100)
{
offset = 1;
DisplayBuffer[0]=0;
goto L0;
}

if(in >= 10)
{
offset = 2;
DisplayBuffer[0]=0;
DisplayBuffer[1]=0;
goto L0;
}

if(in >= 0)
{
offset = 3;
DisplayBuffer[0]=0;
DisplayBuffer[1]=0;
DisplayBuffer[2]=0;
}

L0:
for(counter=offset; counter<4; counter++)
{
DisplayBuffer[counter]=pgm_read_byte(&dec_table[(in/pgm_read_word(&div_table[counter]))%10]); // Адовая конструкция, не так ли?
}

}

int main(void)
{
DDRB  = 0b10111111;

SPSR  = (1<<SPI2X);
SPCR  = (1<<SPE)|(1<<MSTR);

TCCR0 = 0b000000011;
TIMSK = 0b000000001;

asm("sei");

while(1)
{
update();
in++;
_delay_ms(17); // почти 60 кадров
}

return 0;
}
